        function renderBarrages() {
            const grid = document.getElementById('barrages-grid');
            const emptyState = document.getElementById('empty-state');
            const countDisplay = document.getElementById('count-display');
            
            // Apply filter
            if (currentFilter === 'all') {
                filteredBarrages = [...barrages];
            } else {
                filteredBarrages = barrages.filter(b => 
                    b.status.toLowerCase().replace(' ', '-') === currentFilter
                );
            }
            
            // Apply search
            const searchTerm = document.getElementById('search-input').value.toLowerCase();
            if (searchTerm) {
                filteredBarrages = filteredBarrages.filter(b => 
                    b.name.toLowerCase().includes(searchTerm) || 
                    b.region.toLowerCase().includes(searchTerm)
                );
            }
            
            // Apply sort
            filteredBarrages.sort((a, b) => {
                let valA = a[currentSort.field];
                let valB = b[currentSort.field];
                
                if (typeof valA === 'string') valA = valA.toLowerCase();
                if (typeof valB === 'string') valB = valB.toLowerCase();
                
                if (valA < valB) return currentSort.asc ? -1 : 1;
                if (valA > valB) return currentSort.asc ? 1 : -1;
                return 0;
            });
            
            if (filteredBarrages.length === 0) {
                grid.innerHTML = '';
                emptyState.classList.remove('hidden');
            } else {
                emptyState.classList.add('hidden');
                grid.innerHTML = filteredBarrages.map(createCard).join('');
            }
            
            countDisplay.textContent = filteredBarrages.length;
            
            // Update active filter button
            document.querySelectorAll('.filter-btn').forEach(btn => {
                if (btn.dataset.filter === currentFilter) {
                    btn.classList.add('ring-2', 'ring-offset-2', 'ring-blue-500');
                } else {
                    btn.classList.remove('ring-2', 'ring-offset-2', 'ring-blue-500');
                }
            });
        }

        function filterBarrages(filter) {
            currentFilter = filter;
            renderBarrages();
        }

        }

        function sortBarrages(field) {

                currentSort.asc = !currentSort.asc;
            } else {
                currentSort.field = field;
                currentSort.asc = true;
            }

        }

                        ],
                        borderWidth: 0
                    }]
                },
                options: {
                    responsive: true,
                    maintainAspectRatio: false,
                    plugins: {
                        legend: { position: 'bottom' }
                    }
                }
            });
            
            // Region bar chart
            const regionCtx = document.getElementById('regionChart').getContext('2d');
            const regionData = {};
            barrages.forEach(b => {
                if (!regionData[b.region]) regionData[b.region] = { total: 0, count: 0 };
                regionData[b.region].total += b.taux;
                regionData[b.region].count++;
            });
            
            const regions = Object.keys(regionData);
            const avgByRegion = regions.map(r => (regionData[r].total / regionData[r].count).toFixed(1));
            
            new Chart(regionCtx, {
                type: 'bar',
                data: {
                    labels: regions,
                    datasets: [{
                        label: 'Taux moyen (%)',
                        data: avgByRegion,
                        backgroundColor: '#3b82f6',
                        borderRadius: 6
                    }]
                },
                options: {
                    responsive: true,
                    maintainAspectRatio: false,
                    plugins: { legend: { display: false } },
                    scales: {
                        y: { beginAtZero: true, max: 100 }
                    }
                }
            });
        }

        // Export to Excel
        function exportToExcel() {
            const data = barrages.map(b => ({
                'Nom': b.name,
                'Région': b.region,
                'Statut': b.status,
                'Taux de remplissage (%)': b.taux,
                'Volume actuel (Mm³)': b.volume,
                'Capacité max (Mm³)': b.capaciteMax,
                'Hauteur': b.hauteur,
                'Alerte': b.alert ? 'Oui' : 'Non',
                'Description': b.description || ''
            }));
            
            const ws = XLSX.utils.json_to_sheet(data);
            const wb = XLSX.utils.book_new();
            XLSX.utils.book_append_sheet(wb, ws, "Barrages");
            
            // Auto-width columns
            const colWidths = Object.keys(data[0]).map(key => ({
                wch: Math.max(key.length, ...data.map(d => String(d[key]).length)) + 2
            }));
            ws['!cols'] = colWidths;
            
            XLSX.writeFile(wb, `Barrages_Burkina_${new Date().toISOString().split('T')[0]}.xlsx`);
            
            // Show notification
            showNotification('Export Excel réussi !', 'success');
        }

                icon.classList.remove('fa-moon');

        // Service Worker for PWA
        function setupServiceWorker() {
            if ('serviceWorker' in navigator) {
                const swCode = `
                    self.addEventListener('install', e => {
                        e.waitUntil(
                            caches.open('barrages-v1').then(cache => {
                                return cache.addAll(['/']);
                            })
                        );
                    });
                    
                    self.addEventListener('fetch', e => {
                        e.respondWith(
                            caches.match(e.request).then(response => {
                                return response || fetch(e.request);
                            })
                        );
                    });
                `;
                
                const blob = new Blob([swCode], { type: 'application/javascript' });
                const swUrl = URL.createObjectURL(blob);
                
                navigator.serviceWorker.register(swUrl).catch(err => {
                    console.log('Service Worker registration failed');
                });
            }
        }

            if (window.deferredPrompt) {
                window.deferredPrompt.prompt();
                window.deferredPrompt.userChoice.then(choiceResult => {
                    if (choiceResult.outcome === 'accepted') {
                        document.getElementById('pwa-banner').classList.add('hidden');
                    }
                    window.deferredPrompt = null;
                });
            }
        }
