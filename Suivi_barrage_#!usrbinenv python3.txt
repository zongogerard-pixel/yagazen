    def calculate_stats(self):
        """Calcule les statistiques globales"""
        total_taux = sum(b['taux'] for b in self.barrages)
        taux_moyen = round(total_taux / len(self.barrages), 2)
        
        counts = {}
        for b in self.barrages:
            counts[b['status']] = counts.get(b['status'], 0) + 1
            
        return {
            'taux_moyen': taux_moyen,
            'total': len(self.barrages),
            'counts': counts
        }

        function renderBarrages(filter = 'all') {{
            const grid = document.getElementById('barrages-grid');
            const countDisplay = document.getElementById('count-display');
            
            let filtered = barrages;
            if (filter !== 'all') {{
                filtered = barrages.filter(b => b.status.toLowerCase().replace(' ', '-') === filter);
            }}
            
            grid.innerHTML = filtered.map((b, i) => createCard(b, i)).join('');
            countDisplay.textContent = filtered.length;
            
            // Animate progress rings after render
            setTimeout(() => {{
                document.querySelectorAll('.progress-ring-circle').forEach((circle, i) => {{
                    setTimeout(() => {{
                        const targetOffset = circle.getAttribute('stroke-dashoffset');
                        circle.style.strokeDashoffset = targetOffset;
                    }}, i * 50);
                }});
            }}, 100);
            
            // Update active filter button
